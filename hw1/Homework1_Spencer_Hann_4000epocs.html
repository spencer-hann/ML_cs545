Initialize weights vector such that $w_{ji} \in [-0.5, 0.5]$, where $w_{ji}$ is the $i$th weight in the perceptron that predicts if an input image shows the digit $j$ $(0 \le j \le 9)$. For example, if $w_7 \cdot x > w_j$ for all $j \ne 7$, then $7$ is the prediction. More formally, $\hat y = \underset{j}{\operatorname{argmax}} w_j \cdot x$.

import cython
import sys
import numpy as np
cimport numpy as np
import time
from cython.parallel cimport prange


######################################################
##              MATH/UTILITY FUNCTIONS              ##
######################################################

cdef int cargmax(double[10] array) nogil:
    cdef int max = 0
    cdef Py_ssize_t i
    
    for i in range(1,10):
        if max < array[i]:
            max = i
    return max

cdef double dot(double[::1] w, double[::1] x) nogil:
    cdef double dp = 0.0
    cdef Py_ssize_t i
    
    for i in prange(785):
        dp += w[i] * x[i]
        
    return dp


######################################################
##                 ACCURACY TESTING                 ##
######################################################

def predict(w,x):
    cdef double[10] results # C array
    cdef Py_ssize_t j
    
    for j in range(10):
        results[j] = dot(w[j],x)
        
    return cargmax(results)

# very similar to predict, but you know the answer
cdef int check(double[:,::1] w, int t, double[::1] x) nogil:
    cdef double[10] results # C array
    cdef Py_ssize_t j
    
    for j in prange(10):
        results[j] = dot(w[j],x)
        
    return cargmax(results) == t

cdef float test_accuracy(double[:,::1] w, int[::1] T, double[:,::1] X):
    cdef int n = T.shape[0]
    cdef Py_ssize_t i
    cdef int[::1] results = np.empty(n, dtype=np.intc)

    for i in prange(n, nogil=True):
        results[i] = check(w,T[i],X[i])

    return np.sum(results) / float(n)


#######################################################
##                    TRAINING                       ##
#######################################################

cdef void training_example(double[:,::1] w, int t, double[::1] x, double eta) nogil:
    cdef int y
    cdef Py_ssize_t i, j
    cdef double delta_w
    
    for j in prange(10):
        y = 1 if dot(w[j],x) > 0 else 0
        delta_w = eta*(1-y) if j == t else eta*(0-y)
        if not delta_w: continue
        for i in prange(785):
            w[j,i] += delta_w * x[i]

cdef np.float train_epoch(double[:,::1] w, int[::1] T, double[:,::1] X, double eta):
    cdef int i, n = T.shape[0]
    
    for i in prange(n, nogil=True):
        training_example(w, T[i], X[i],eta)
    return test_accuracy(w,T,X)

def train(
    double[:,:,::1] w, 
    int[::1] T_train, int[::1] T_test, 
    double[:,::1] X_train, double[:,::1] X_test,
    double[:,::1] training_accuracies,
    double[:,::1] testing_accuracies, 
    int n=50, np.float eta=1.
):
    cdef Py_ssize_t i
    cdef list max_test = [0.0,0.0,0.0]

    for i in range(3):
        # epoch 0 accuracies
        training_accuracies[i,0] = test_accuracy(w[i],T_train,X_train)
        testing_accuracies[i,0] = test_accuracy(w[i],T_test,X_test)
        
        # training phase n epochs
        for j in range(1,n):
            training_accuracies[i,j] = train_epoch(w[i],T_train,X_train,eta)
            testing_accuracies[i,j] = test_accuracy(w[i],T_test,X_test)

            if testing_accuracies[i,j] > max_test[i]:
                max_test[i] = testing_accuracies[i,j]
                
        eta /= 10
        print(f"n{eta} accuracy :", training_accuracies[i,n-1], testing_accuracies[i,n-1])
        print(time.ctime())
    
    return "Best for each 'eta':", max_test

def create_conf_matrix(w, T, X):
    N = T.shape[0]
    results = np.zeros((3,11,11), dtype=np.intc)

    for l in range(3):
        for n in range(N):
            i = T[n]
            j = predict(w[l],X[n])
            results[l,i,j] += 1
            results[l,i,10] += 1
            results[l,10,j] += 1

    return results

def display_conf_matrix(conf_matrix):
    _,ax1 = plt.subplots(figsize=(10,10))
    _,ax2 = plt.subplots(figsize=(10,10))
    ax1.set_title("Count Confusion Matrix")
    ax2.set_title("Percentage Confusion Matrix")

    ratio_conf_matrix = np.zeros(conf_matrix.shape, dtype=np.float)

    for i in range(conf_matrix.shape[0]):
        for j in range(conf_matrix.shape[1]):
            if i == 10 and j == 10: break
            num = conf_matrix[i,j]
            ax1.text(i, j, str(num), va="center", ha="center")
            if j == 10 or i == 10: continue
            ratio_conf_matrix[i,j] = conf_matrix[i,j] / float(conf_matrix[10,j])
            num = ratio_conf_matrix[i,j] * 100
            ax2.text(i, j, "%.1f%%" % num, va="center", ha="center")

    ax1.matshow(conf_matrix[:10,:10], cmap=plt.cm.Wistia)
    ax2.matshow(ratio_conf_matrix[:10,:10], cmap=plt.cm.Wistia)
    
for matrix in create_conf_matrix(w, t_testing, testing):
    display_conf_matrix(matrix)
